``` c
#include <stdio.h>

int main() {
    int attempts = 0;  // 尝试次数
    int success = 0;   // 是否成功

    printf("人生就像编程，每一次尝试都让我们更接近目标...\n");

    while (!success && attempts < 5) {
        printf("第 %d 次尝试...\n", attempts + 1);
        attempts++;

        // 假设每一次都会进步
        if (attempts == 5) {
            success = 1;  // 最后一次成功
        }
    }

    if (success) {
        printf("成功了！每一次失败，都是走向成功的一步！\n");
    } else {
        printf("继续前行，不要停下。\n");
    }

    printf("记住：只要你不放弃，成功永远是可能的！\n");

    return 0;
}
```
